  }

  // 1. 获取用户的活跃关键词
  const keywords = await Keyword.find({ userId, isActive: true });

  // 2. 调用爬虫服务
  const crawlResult = await crawlAllSources(keywordNames, {
    sources: sources || ['hackernews', 'google', 'bing', 'ddg', 'sogou'],
    maxResultsPerSource: limit || 20
  });

  // 3. 保存到数据库（启用AI分析）
  for (const hotspot of crawlResult.results) {
    const aiAnalysis = await quickAnalyzeWithAI(hotspot.content, hotspot.source);

    const newHotspot = new Hotspot({
      userId: userId,
      title: hotspot.title,
      content: hotspot.content,
      isVerified: aiAnalysis.isAuthentic,
      confidence: aiAnalysis.confidence,
      keywords: hotspot.keywords
    });
    await newHotspot.save();
  }

  // 4. 返回结果
  res.json({ success: true, saved: savedHotspots.length });
};

async function crawlAllSources(keywords = [], options = {}) {
  const allResults = [];

  // 并发爬取多个信息源
  if (sources.includes('hackernews')) {
    const hnResults = await hackernewsCrawler.crawlHackerNews(keywords, maxResultsPerSource);
    allResults.push(...hnResults);
  }

  if (sources.includes('bing')) {
    const bingResults = await bingCrawler.crawlBing(keywords, options);
    allResults.push(...bingResults);
  }

  // ... 其他信息源

  // URL去重
  const uniqueResults = removeDuplicates(allResults);

  return {
    results: uniqueResults,
    totalFetched: allResults.length,
    totalUnique: uniqueResults.length
  };
}

exports.quickAnalyzeWithAI = async (content, source) => {
  const provider = getAIProvider(); // GLM / DeepSeek / OpenRouter

  const prompt = `请判断以下内容的真实性并评分（0-100分）。
内容：${content.substring(0, 150)}
来源：${source}

要求：直接返回JSON格式
{"isAuthentic":true,"confidence":75,"reason":"简短理由","summary":"内容摘要"}`;

  const response = await axios.post(provider.apiUrl, {
    model: provider.model,  // glm-4.7
    messages: [{ role: 'user', content: prompt }],
    max_tokens: 300,
    temperature: 0.1
  }, { headers: provider.headers, timeout: 35000 });

  return {
    isAuthentic: result.isAuthentic,
    confidence: result.confidence,
    reasons: [result.reason],
    summary: result.summary
  };
};

const fetchHotspots = async () => {
  const response = await api.get('/api/content/hotspots', {
    params: { limit: 50, offset: 0 }
  });

  const hotspotsData = response.data?.data || [];

  // 更新状态
  setHotspots(hotspotsData);

  // 计算统计数据
  const todayHotspots = hotspotsData.filter(h =>
    new Date(h.timestamp) >= today
  ).length;
  const verifiedContent = hotspotsData.filter(h => h.isVerified).length;

  setStats(prev => ({
    ...prev,
    todayHotspots,
    verifiedContent
  }));
};
